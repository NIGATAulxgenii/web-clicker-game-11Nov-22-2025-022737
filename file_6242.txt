            const messageElement = document.getElementById('dynamicMessage');
            if (messageElement.textContent === 'This is a simple web page. Click the button below to see some interaction!') {
                messageElement.textContent = 'Wow! You just changed the text. Isn\'t that neat?';
            } else {
